    <script>
        let currentPage = 1;
        const totalPages = 6;

        function changePage(page) {
            // Ensure page is within valid range
            if (page < 1 || page > totalPages) return;
            
            currentPage = page;
            const cards = document.querySelectorAll('.carde');
            const buttons = document.querySelectorAll('.paginations button');
            const backBtn = document.getElementById('backBtn');
            const nextBtn = document.getElementById('nextBtn');

            // Update card visibility
            cards.forEach(card => {
                card.classList.remove('active');
                if (parseInt(card.getAttribute('data-page')) === currentPage) {
                    card.classList.add('active');
                }
            });

            // Update button active state
            buttons.forEach(button => {
                button.classList.remove('active');
                if (button.textContent !== 'Back' && button.textContent !== 'Next' && parseInt(button.textContent) === currentPage) {
                    button.classList.add('active');
                }
            });

            // Enable/disable Back and Next buttons
            backBtn.disabled = currentPage === 1;
            nextBtn.disabled = currentPage === totalPages;
        }

        // Initialize the first page
        changePage(1);
    </script>
